---
title: 自定义事件
date: 2025-08-04 10:04:34
tags: JS
---

```tsx
function App() {
  const myEvent = new CustomEvent("my-event", {
    detail: {
      data: [1, 2, 3],
    },
  });
  window.addEventListener("my-event", (e) => {
    console.log("e", e);
  });
  return (
    <>
      <div className="card">
        <button onClick={() => window.dispatchEvent(myEvent)}>
          触发自定义事件
        </button>
      </div>
    </>
  );
}

export default App;
```
